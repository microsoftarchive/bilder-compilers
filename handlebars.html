  var template = [
    "define(['vendor/handlebars'], function(H) {",
      "var name = '%s';",
      "return H.template(%s);",
    "});"
  ].join('\n');

  function name (file, options) {
    var prefixRegexp = new RegExp('^' + options.src + '/');
    return file.replace(prefixRegexp, '').replace(suffixRegExp, '');
  }
